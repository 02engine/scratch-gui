```diff
+        handleToolboxRequest: () => blockDisableMenu.handleToolboxRequest()
+    });

+    console.log('Block disable extension initialized');
```

```diff
+            this.vm.emit('BLOCK_DISABLE_CHANGED', {type: 'category', id: categoryId, disabled: false});
+        });
+    }
+}
```

```diff
diff --git a/src/lib/vm-block-disable.js b/src/lib/vm-block-disable.js

+import VM from 'scratch-vm';
+
+/**
+ * VM Block Disable - Provides functionality to disable blocks at the VM level
+ */
+class VMBlockDisable {
+    constructor (vm) {
+        this.vm = vm;
+        this.disabledBlocks = new Set();
+        this.disabledCategories = new Set();
+        this.originalCreateBlock = null;
+        this.originalExecute = null;
+    }
+
+    /**
+     * Initialize the block disable functionality
+     */
+    initialize () {
+        // Store original methods
+        this.originalCreateBlock = this.vm.runtime.createBlock;
+        this.originalExecute = this.vm.runtime.sequencer.execute;
+
+        // Override createBlock to check for disabled blocks
+        this.vm.runtime.createBlock = (block, ...args) => {
+            const blockId = block.id || block;
+            if (this.disabledBlocks.has(blockId)) {
+                console.warn(`Block ${blockId} is disabled and cannot be created`);
+                return null;
+            }
+            return this.originalCreateBlock.call(this.vm.runtime, block, ...args);
+        };
+
+        // Override execute to check for disabled blocks
+        this.vm.runtime.sequencer.execute = (...args) => {
+            const thread = args[0];
+            if (thread && thread.topBlock) {
+                const blockId = thread.topBlock;
+                if (this.disabledBlocks.has(blockId)) {
+                    console.warn(`Block ${blockId} is disabled and cannot be executed`);
+                    return Promise.resolve();
+                }
+            }
+            return this.originalExecute.call(this.vm.runtime.sequencer, ...args);
+        };
+
+        // Listen for block disable changes
+        this.vm.on('BLOCK_DISABLE_CHANGED', (data) => {
+            if (data.type === 'block') {
+                if (data.disabled) {
+                    this.disabledBlocks.add(data.id);
+                } else {
+                    this.disabledBlocks.delete(data.id);
+                }
+            } else if (data.type === 'category') {
+                if (data.disabled) {
+                    this.disabledCategories.add(data.id);
+                } else {
+                    this.disabledCategories.delete(data.id);
+                }
+            }
+        });
+    }
+
+    /**
+     * Restore original VM methods
+     */
+    cleanup () {
+        if (this.originalCreateBlock) {
+            this.vm.runtime.createBlock = this.originalCreateBlock;
+        }
+        if (this.originalExecute) {
+            this.vm.runtime.sequencer.execute = this.originalExecute;
+        }
+    }
+}
+
+/**
+ * Initialize VM block disable functionality
+ * @param {VM} vm - The Scratch VM instance
+ * @returns {VMBlockDisable} - The VM block disable instance
+ */
+const initializeVMBlockDisable = vm => {
+    const blockDisable = new VMBlockDisable(vm);
+    blockDisable.initialize();
+    return blockDisable;
+};
+
+export default initializeVMBlockDisable;
```
